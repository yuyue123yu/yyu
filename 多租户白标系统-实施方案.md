### 1.1 数据隔离 RLS 策略

#### 步骤 1：审计现有表结构
需要添加 `tenant_id` 的表：

- ❓ `consultations` - 需要检查

- ❓ `articles` - 需要检查
- ❓ `templates` - 需要检查
- ❓ `orders` - 需要检查

#### 步骤 2：创建 RLS 辅助函数
```sql
-- 获取当前用户的 tenant_id
CREATE OR REPLACE FUNCTION auth.current_user_tenant_id()
RETURNS UUID AS $$
  SELECT tenant_id FROM public.profiles WHERE id = auth.uid()
$$ LANGUAGE SQL STABLE;

-- 检查是否为 Super Admin
CREATE OR REPLACE FUNCTION auth.is_super_admin()
RETURNS BOOLEAN AS $$
  SELECT COALESCE(
    (SELECT super_admin FROM public.profiles WHERE id = auth.uid()),
    false
  )
$$ LANGUAGE SQL STABLE;
```

#### 步骤 3：配置 RLS 策略模板
```sql
-- 示例：consultations 表的 RLS 策略
-- SELECT: 只能查看自己租户的数据（Super Admin 除外）
CREATE POLICY "tenant_isolation_select" ON consultations
  FOR SELECT
  TO authenticated
  USING (
    auth.is_super_admin() OR 
    tenant_id = auth.current_user_tenant_id()
  );

-- INSERT: 只能插入到自己的租户
CREATE POLICY "tenant_isolation_insert" ON consultations
  FOR INSERT
  TO authenticated
  WITH CHECK (
    tenant_id = auth.current_user_tenant_id()
  );

-- UPDATE: 只能更新自己租户的数据
CREATE POLICY "tenant_isolation_update" ON consultations
  FOR UPDATE
  TO authenticated
  USING (
    tenant_id = auth.current_user_tenant_id()
  )
  WITH CHECK (
    tenant_id = auth.current_user_tenant_id()
  );

-- DELETE: 只能删除自己租户的数据
CREATE POLICY "tenant_isolation_delete" ON consultations
  FOR DELETE
  TO authenticated
  USING (
    tenant_id = auth.current_user_tenant_id()
  );
```

#### 步骤 4：Super Admin 特殊处理
```sql
-- Super Admin 可以访问所有租户数据
CREATE POLICY "super_admin_full_access" ON consultations
  FOR ALL
  TO authenticated
  USING (auth.is_super_admin())
  WITH CHECK (auth.is_super_admin());
```

### 1.2 Middleware 租户识别

#### 文件：`middleware.ts`
```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';
import { createClient } from '@/lib/supabase/middleware';

export async function middleware(request: NextRequest) {
  const { pathname, hostname } = request.nextUrl;
  
  // 跳过 API 路由、静态文件、Super Admin
  if (
    pathname.startsWith('/api') ||
    pathname.startsWith('/_next') ||
    pathname.startsWith('/super-admin') ||
    pathname.startsWith('/admin')
  ) {
    return NextResponse.next();
  }

  // 1. 识别租户
  let tenantSubdomain: string | null = null;
  let tenantCustomDomain: string | null = null;

  // 检查是否为子域名
  const hostParts = hostname.split('.');
  if (hostParts.length >= 3 && hostParts[0] !== 'www') {
    // 例如：tenant1.legalmy.com
    tenantSubdomain = hostParts[0];
  } else if (hostname !== 'legalmy.com' && hostname !== 'localhost') {
    // 自定义域名
    tenantCustomDomain = hostname;
  }

  // 如果没有识别到租户，返回主站
  if (!tenantSubdomain && !tenantCustomDomain) {
    return NextResponse.next();
  }

  // 2. 查询租户信息
  const supabase = createClient(request);
  
  let query = supabase
    .from('tenants')
    .select('*')
    .eq('status', 'active')
    .single();

  if (tenantSubdomain) {
    query = query.eq('subdomain', tenantSubdomain);
  } else if (tenantCustomDomain) {
    query = query.eq('custom_domain', tenantCustomDomain);
  }

  const { data: tenant, error } = await query;

  // 3. 租户不存在或已停用
  if (error || !tenant) {
    return NextResponse.rewrite(new URL('/404', request.url));
  }

  // 4. 将租户信息注入到请求头
  const response = NextResponse.next();
  response.headers.set('x-tenant-id', tenant.id);
  response.headers.set('x-tenant-subdomain', tenant.subdomain);
  response.headers.set('x-tenant-branding', JSON.stringify(tenant.branding || {}));
  response.headers.set('x-tenant-features', JSON.stringify(tenant.features || {}));

  return response;
}

export const config = {
  matcher: [
    /*
     * Match all request paths except:
     * - _next/static (static files)
     * - _next/image (image optimization files)
     * - favicon.ico (favicon file)
     * - public folder
     */
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
};
```

#### 创建 Middleware Supabase Client
**文件：`src/lib/supabase/middleware.ts`**
```typescript
import { createServerClient, type CookieOptions } from '@supabase/ssr';
import { NextRequest, NextResponse } from 'next/server';

export function createClient(request: NextRequest) {
  let response = NextResponse.next({
    request: {
      headers: request.headers,
    },
  });

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get(name: string) {
          return request.cookies.get(name)?.value;
        },
        set(name: string, value: string, options: CookieOptions) {
          request.cookies.set({
            name,
            value,
            ...options,
          });
          response = NextResponse.next({
            request: {
              headers: request.headers,
            },
          });
          response.cookies.set({
            name,
            value,
            ...options,
          });
        },
        remove(name: string, options: CookieOptions) {
          request.cookies.set({
            name,
            value: '',
            ...options,
          });
          response = NextResponse.next({
            request: {
              headers: request.headers,
            },
          });
          response.cookies.set({
            name,
            value: '',
            ...options,
          });
        },
      },
    }
  );

  return supabase;
}
```

### 1.3 TenantContext 全局状态

#### 文件：`src/contexts/TenantContext.tsx`

```typescript
'use client';

import React, { createContext, useContext, useEffect, useState } from 'react';

interface TenantBranding {
  logo_url: string;
  primary_color: string;
  secondary_color: string;
  company_name: string;
  tagline: string;
}

interface TenantFeatures {
  consultations: boolean;
  lawyers: boolean;
  articles: boolean;
  templates: boolean;
}

interface TenantContactInfo {
  email: string;
  phone: string;
  address: string;
}

interface Tenant {
  id: string;
  name: string;
  subdomain: string;
  custom_domain: string | null;
  status: string;
  branding: TenantBranding;
  features: TenantFeatures;
  contact_info: TenantContactInfo;
}

interface TenantContextType {
  tenant: Tenant | null;
  isLoading: boolean;
  error: string | null;
  refreshTenant: () => Promise<void>;
}

const TenantContext = createContext<TenantContextType | undefined>(undefined);

export function TenantProvider({ children }: { children: React.ReactNode }) {
  const [tenant, setTenant] = useState<Tenant | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchTenant = async () => {
    try {
      setIsLoading(true);
      setError(null);

      // 从 API 获取当前租户信息
      const response = await fetch('/api/tenant/current');
      
      if (!response.ok) {
        throw new Error('Failed to fetch tenant');
      }

      const data = await response.json();
      
      if (data.success && data.tenant) {
        setTenant(data.tenant);
      } else {
        setTenant(null);
      }
    } catch (err: any) {
      console.error('Error fetching tenant:', err);
      setError(err.message);
      setTenant(null);
    } finally {
      setIsLoading(false);
    }
  };

  useEffect(() => {
    fetchTenant();
  }, []);

  const refreshTenant = async () => {
    await fetchTenant();
  };

  return (
    <TenantContext.Provider value={{ tenant, isLoading, error, refreshTenant }}>
      {children}
    </TenantContext.Provider>
  );
}

export function useTenant() {
  const context = useContext(TenantContext);
  if (context === undefined) {
    throw new Error('useTenant must be used within a TenantProvider');
  }
  return context;
}
```

#### API Route：`src/app/api/tenant/current/route.ts`
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createServerClient } from '@/lib/supabase/server';

export async function GET(request: NextRequest) {
  try {
    // 从请求头获取租户 ID（由 middleware 注入）
    const tenantId = request.headers.get('x-tenant-id');

    if (!tenantId) {
      return NextResponse.json({
        success: false,
        error: 'No tenant identified',
      });
    }

    // 查询租户信息
    const supabase = await createServerClient();
    const { data: tenant, error } = await supabase
      .from('tenants')
      .select('*')
      .eq('id', tenantId)
      .eq('status', 'active')
      .single();

    if (error || !tenant) {
      return NextResponse.json({
        success: false,
        error: 'Tenant not found',
      });
    }

    return NextResponse.json({
      success: true,
      tenant,
    });
  } catch (error: any) {
    console.error('Error fetching tenant:', error);
    return NextResponse.json({
      success: false,
      error: error.message,
    }, { status: 500 });
  }
}
```

### 1.4 安全验证机制

#### API 层 Double-Check
**所有 API Route 都要验证 tenant_id**

```typescript
// 示例：src/app/api/consultations/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { createServerClient } from '@/lib/supabase/server';

export async function GET(request: NextRequest) {
  try {
    const supabase = await createServerClient();
    
    // 1. 验证用户登录
    const { data: { session } } = await supabase.auth.getSession();
    if (!session) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    // 2. 获取用户的 tenant_id
    const { data: profile } = await supabase
      .from('profiles')
      .select('tenant_id, super_admin')
      .eq('id', session.user.id)
      .single();

    if (!profile) {
      return NextResponse.json({ error: 'Profile not found' }, { status: 404 });
    }

    // 3. 构建查询（Super Admin 可以看所有数据）
    let query = supabase.from('consultations').select('*');
    
    if (!profile.super_admin) {
      // 普通用户只能看自己租户的数据
      query = query.eq('tenant_id', profile.tenant_id);
    }

    const { data: consultations, error } = await query;

    if (error) {
      throw error;
    }

    return NextResponse.json({
      success: true,
      consultations,
    });
  } catch (error: any) {
    console.error('Error fetching consultations:', error);
    return NextResponse.json({
      success: false,
      error: error.message,
    }, { status: 500 });
  }
}
```

#### Session 与 TenantContext 联动
**在 Layout 中验证**

```typescript
// src/app/(public)/layout.tsx
'use client';

import { useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { useTenant } from '@/contexts/TenantContext';
import { useAuth } from '@/contexts/AuthContext';

export default function PublicLayout({ children }: { children: React.ReactNode }) {
  const router = useRouter();
  const { tenant, isLoading: tenantLoading } = useTenant();
  const { user, profile, isLoading: authLoading } = useAuth();

  useEffect(() => {
    // 等待加载完成
    if (tenantLoading || authLoading) return;

    // 如果用户已登录，验证 tenant_id 是否匹配
    if (user && profile && tenant) {
      if (profile.tenant_id !== tenant.id && !profile.super_admin) {
        // tenant_id 不匹配，强制登出
        console.error('Tenant mismatch detected');
        router.push('/logout');
      }
    }
  }, [tenant, user, profile, tenantLoading, authLoading, router]);

  return <>{children}</>;
}
```

### 2.2 CSS Variables 动态主题

```typescript
'use client';

import { useTenant } from '@/contexts/TenantContext';
import { useEffect } from 'react';

export default function PublicLayout({ children }: { children: React.ReactNode }) {
  const { tenant, isLoading } = useTenant();

  useEffect(() => {
    if (tenant?.branding) {
      // 注入 CSS Variables
      document.documentElement.style.setProperty(
        '--color-primary',
        tenant.branding.primary_color || '#FF6B35'
      );
      document.documentElement.style.setProperty(
        '--color-secondary',
        tenant.branding.secondary_color || '#004E89'
      );
    }
  }, [tenant]);

  if (isLoading) {
    return <div>Loading...</div>;
  }

  return <>{children}</>;
}
```

```typescript
import { useTenant } from '@/contexts/TenantContext';

export function useFeature(featureName: keyof TenantFeatures): boolean {
  const { tenant } = useTenant();
  
  if (!tenant?.features) {
    return false;
  }
  
  return tenant.features[featureName] ?? false;
}
```
